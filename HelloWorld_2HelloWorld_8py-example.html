20import os
21import SimpleITK as sitk

23# Create an image
24pixelType = sitk.sitkUInt8
25imageSize = [128, 128]
26image = sitk.Image(imageSize, pixelType)

28# Create a face image
29faceSize = [64, 64]
30faceCenter = [64, 64]
31face = sitk.GaussianSource(pixelType, imageSize, faceSize, faceCenter)
32
33# Create eye images
34eyeSize = [5, 5]
35eye1Center = [48, 48]
36eye2Center = [80, 48]
37eye1 = sitk.GaussianSource(pixelType, imageSize, eyeSize, eye1Center, 150)
38eye2 = sitk.GaussianSource(pixelType, imageSize, eyeSize, eye2Center, 150)
39
40# Apply the eyes to the face
41face = face - eye1 - eye2
42face = sitk.BinaryThreshold(face, 200, 255, 255)
43
44# Create the mouth
45mouthRadii = [30, 20]
46mouthCenter = [64, 76]
47mouth = 255 - sitk.BinaryThreshold(sitk.GaussianSource(
48 pixelType, imageSize, mouthRadii, mouthCenter),
49 200, 255, 255)
50# Paste the mouth into the face
51mouthSize = [64, 18]
52mouthLoc = [32, 76]
53face = sitk.Paste(face, mouth, mouthSize, mouthLoc, mouthLoc)
54
55# Apply the face to the original image
56image = image + face
57

59if ("SITK_NOSHOW" not in os.environ):
