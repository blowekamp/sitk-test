28/* \brief An object for selecting an integer value at compile-time.
29 *
30 *
31 * If VCode is true, then the member Value will be TIfTrue, otherwise
32 * the member Value will be equivalent to TIfFalse.
33 *
34 * This class also contains a typedef for ValueType, which is an the
35 * selected integral constant as a type.
36 */
37template <bool VCond, int TIfTrue, int TIfFalse>
38struct ConditionalValue
39{
40private:
41 using Self = ConditionalValue;
42 typedef typename std::conditional<VCond,
43 std::integral_constant<int, TIfTrue>,
44 std::integral_constant<int, TIfFalse> >::type ConditionalType;
45public:
46 using Type = typename ConditionalType::type;
47 using ValueType = typename ConditionalType::value_type;
48
49 static const ValueType Value = ConditionalType::value;
50};
